확인 버튼을 누르면 1~100까지 더한 값이 얼럿창에 표시 됩니다.

문제3.1부터 50까지 369결과를 프린트하자.

*아래의 코드를 참고하여 369 결과를 알 수 있게 했습니다.

let num = 30
if (!num || num < 1 || num > 50) {
console.log("1~50 사이의 숫자를 입력해주세요.");
}

let strNum = String(num);
let result = 0;

for (let i of strNum) {
if (i === "3" || i === "6" || i === "9") {
result++;
}
}

if (result === 0) {
console.log(num);
} else {
console.log("짝".repeat(result));
}
